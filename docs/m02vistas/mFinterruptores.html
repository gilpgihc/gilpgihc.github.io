  if (/.*android.*/i.
   test(navigator.userAgent)) {
   document.body.classList
    .add("android");
  }
  if (
   /.*(iPad|iPhone|iPod|Mac).*/.
    test(navigator.userAgent)) {

    add('ios');

  class CmInt extends HTMLElement {
   connectedCallback() {
    setTimeout(
     () => this.prepara(),
     100);
   }
   prepara() {
    /** @private */
    this._input =
     this.querySelector("input");
    const i = this._input;
    if (i) {
     i.type = "range";
     i.min = "0";
     i.max = "1";
     i.addEventListener("change",
      () => this.analiza());
     this.analiza();
    }
   }
   analiza() {
    const i = this._input;
    if (i) {
     if (i.valueAsNumber === 0) {
      i.classList.add("off");
     } else {
      i.classList.remove("off");
     }
    }
   }
  }
