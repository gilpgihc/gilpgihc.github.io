P. Cajón de navegación

  • Se usa para seleccionar entre 5 o más vistas.

  • Se usa en:

    • Móvil

    • Tablet

    • Escritorio

  • Algunos diseñadores prefieren usar pestañas, barras de navegación o navigation rail.

  • Se crea el elemento personalizado mi-nav, que puede usarse en todas las vistas de la app, colocándose dentro de una cn-app-bar.

Salida

Ábrelo en otra pestaña.

 <script>
  const CLASE_ELEVACION2 =
   "elevacion2"

  class CnAppBar
   extends HTMLElement {
   constructor() {
    super()
    /** @type {number} */
    this.posY = 0
    /** @type {boolean} */
    this.procesando = false
   }
   connectedCallback() {
    window.addEventListener(
     "scroll",
     () => this.onScroll())
   }
   onScroll() {
    this.posY = window.scrollY
    if (!this.procesando) {
     window.requestAnimationFrame(
      () => this.avanza())
    }
    this.procesando = true
   }
   avanza() {
    if (this.posY === 0) {
     this.classList.
      remove(CLASE_ELEVACION2)
    } else {
     this.classList.
      add(CLASE_ELEVACION2)
    }
    this.procesando = false
   }
  }
  customElements.
   define("cn-app-bar", CnAppBar)
