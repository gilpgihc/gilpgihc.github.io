• Se usa para seleccionar entre 2 o más vistas.

  • Se usa en:

    • Móvil

    • Tablet

    • Escritorio

  • Se crea el elemento personalizado mi-nav, que puede usarse en todas las vistas de la app.

  • En el método connectedCallback, añade la instrucción this.classList.add("navTab"), para indicar que usa pestañas.

  • Coloca mi-nav, usando el atributo slot="adicional". dentro de una barra de aplicación.

Salida

Ábrelo en otra pestaña.

Revísalo en gilpgedit.

     querySelector(

     this._avatar.addEventListener(
      "slotchange",

   }

    if (this._avatar !== null) {
     for (const e of this._avatar
      .assignedElements()) {

       e.querySelector("span")
      if (span !== null) {

        "var(--OnSurfaceVariant)"
       span.style.padding = "5px"

      }
     }
    }
   }

   /** @private */
   _onScroll() {
    this._posY = scrollY
    if (!this._scrolling) {
     requestAnimationFrame(
      () => this._avanza())
    }
    this._scrolling = true
   }

   /** @private */
   _avanza() {
    if (this._posY === 0) {
     this.classList.
      remove(ELEVATION_CLASS)
    } else {
     this.classList.
      add(ELEVATION_CLASS)
    }
    this._scrolling = false
   }
  }

   "center-aligned-top-app-bar",
